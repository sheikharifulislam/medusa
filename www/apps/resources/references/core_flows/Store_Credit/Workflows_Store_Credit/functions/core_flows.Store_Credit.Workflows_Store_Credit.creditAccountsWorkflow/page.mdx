# creditAccountsWorkflow - Medusa Core Workflows Reference

This documentation provides a reference to the `creditAccountsWorkflow`. It belongs to the `@medusajs/medusa/core-flows` package.

This workflow credits one or more store credit accounts with transactions.

You can use this workflow within your own customizations or custom workflows,
allowing you to wrap custom logic around crediting store credit accounts.

```ts title="src/api/workflow/route.ts"
import type {
  MedusaRequest,
  MedusaResponse,
} from "@medusajs/framework/http"
import { creditAccountsWorkflow } from "@medusajs/loyalty-plugin/workflows"

export async function POST(
  req: MedusaRequest,
  res: MedusaResponse
) {
  const { result } = await creditAccountsWorkflow(req.scope)
    .run({
      input: [{
        account_id: "sca_123",
        amount: 100,
        reference: "gift_card",
        reference_id: "gc_123",
      }, ],
    })

  res.send(result)
}
```
